1. 实例一、一个线程依赖于另外几个线程

只有当所依赖的线程执行完毕后，当前线程才会继续执行。

1.1. 原始实现

public class ThreadEndWait {

    private int flag = 0;

    public static void main(String[] args) throws InterruptedException {
        ThreadEndWait threadEndWait = new ThreadEndWait();
        final int threadNum = 10;

        for (int i = 0; i < threadNum; i++) {
            new Thread(new MyRunnable(i, threadEndWait)).start();
        }

        while (true) {
            if (threadEndWait.flag >= threadNum) {
                System.out.println("所有的线程都执行完毕");
                break;
            }
            Thread.sleep(10);
        }
    }


    static class MyRunnable implements Runnable {

        int threadIndex;
        ThreadEndWait threadEndWait;

        public MyRunnable(int index, ThreadEndWait threadEndWait) {
            this.threadIndex = index;
            this.threadEndWait = threadEndWait;
        }

        public void run() {
            try {
                long nextLong = new Random().nextInt(999);
                Thread.sleep(nextLong);
                System.out.println("线程" + threadIndex + "执行完毕，耗时" + nextLong + "毫秒");
                synchronized (threadEndWait) {
                    threadEndWait.flag++;
                }
                System.out.println("线程 " + threadIndex + " => " + threadEndWait.flag);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}

1.2. 使用FutureTask

public class FutureTaskEndWait {

    public static void main(String[] args) throws ExecutionException, InterruptedException {

        FutureTask<String>[] tasks = new FutureTask[10];
        for (int i = 0; i < tasks.length; i++) {
            final int j = i;
            tasks[i] = new FutureTask<String>(new Callable<String>() {
                public String call() throws Exception {
                    long nextLong = new Random().nextInt(999);
                    Thread.sleep(nextLong);
                    return "线程" + j + "执行完毕，耗时" + nextLong + "毫秒";
                }
            });
            new Thread(tasks[i]).start();
        }

        for (int i = 0; i < tasks.length; i++) {
            System.out.println(tasks[i].get());
        }

        System.out.println("所有的任务执行完成");
    }
}

1.3. 使用CyclicBarrier

public class CyclicBarrierEndWait {
    public static void main(String[] args) throws Exception {

        final int threadNum = 10;
        //注意：10个子线程 + 1个主线程
        CyclicBarrier cb = new CyclicBarrier(threadNum + 1);

        for (int i = 0; i < threadNum; i++) {
            new Thread(new MyRunable(cb, i)).start();
        }

        cb.await();
        System.out.println("-----------\n所有thread执行完成！");
    }

    static class MyRunable implements Runnable {
        CyclicBarrier _cb;
        int _i;

        public MyRunable(CyclicBarrier cb, int i) {
            this._cb = cb;
            this._i = i;
        }

        public void run() {
            try {
                Thread.sleep((long) (Math.random() * 100));
                System.out.println("thread " + _i + " done，正在等候其它线程完成...");
                _cb.await();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }
}

2. 实例二、火车票售卖

火车票资源类

public class Resource {
    int ticketNum = 100;
    Boolean ticketEmpty = false;

    public Resource() {}

    public Resource(int ticketNum) {
        this.ticketNum = ticketNum;
    }

    public synchronized void sellTicket(Seller seller) {
        if (ticketNum > 0) {
            System.out.println("第" + seller.sellerIndex + "售票点卖出了第" + ticketNum + "张票……");
            ticketNum--;
        } else {
            ticketEmpty = true;
        }
    }
}

火车站窗口类

public class Seller implements Runnable{

    int sellerIndex;
    Resource resource;

    public Seller(int sellerIndex, Resource resource) {
        this.sellerIndex = sellerIndex;
        this.resource = resource;
    }

    public void run() {
        while (!resource.ticketEmpty) {
            resource.sellTicket(this);

            try {
                long nextLong = new Random().nextInt(999);
                Thread.sleep(nextLong);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}

启动类

public class SellTicket {

    public static void main(String[] args) {
        Resource resource = new Resource(10);

        for (int i = 1; i < 11; i++) {
            new Thread(new Seller(i, resource)).start();
        }
    }
}

3. 实例三、ATM取钱

银行账号类

public class Account {

    private Integer accountNum;

    private Double balance;

    public Account(Integer accountNum, Double balance) {
        this.accountNum = accountNum;
        this.balance = balance;
    }

    public Account() {
    }

    public Integer getAccountNum() {
        return accountNum;
    }

    public void setAccountNum(Integer accountNum) {
        this.accountNum = accountNum;
    }

    public Double getBalance() {
        return balance;
    }

    public void setBalance(Double balance) {
        this.balance = balance;
    }
}

取钱操作类

public class WithDrawAction implements Runnable {

    private String name;

    private Account account;

    private Double withDrawBalance;

    public WithDrawAction(String name, Account account, Double withDrawBalance) {
        this.name = name;
        this.account = account;
        this.withDrawBalance = withDrawBalance;
    }

    public void run() {
        synchronized (account) {
            if (withDrawBalance <= account.getBalance()) {
                System.out.println(name + "从" + account.getAccountNum() + "中成功取走" + withDrawBalance + "元");
                account.setBalance(account.getBalance() - withDrawBalance);
            } else {
                System.out.println(name + "从" + account.getAccountNum() + "中取" + withDrawBalance + "元失败，余额不足！");
            }
        }
    }
}

模拟取钱

public class BankATM {

    public static void main(String[] args) {
        Account account = new Account(9999, 1000.0);

        new WithDrawAction("Tom", account, 600.0).run();
        new WithDrawAction("Jerry", account, 600.0).run();
    }
}

4. 实例四、龟兔赛跑

动物抽象类

public abstract class Animal extends Thread {

    /**
     * 比赛长度
     */
    public double raceLength = 20.0;

    /**
     * 回调
     */
    public RaceCallback raceCallback;

    /**
     * 跑
     */
    public abstract void running();

    @Override
    public void run() {
        super.run();
        while (raceLength > 0) {
            running();
        }
    }
}

回调接口

public interface RaceCallback {

    void win(Animal animal) throws InterruptedException;
}

兔子

public class Rabbit extends Animal {

    public Rabbit() {
        setName("Rabbit");
    }

    @Override
    public void running() {
        try {
            while (!Thread.currentThread().isInterrupted()) {
                System.out.println(this.getName() + Thread.currentThread().isInterrupted());
                double stepLength = 0.5;

                raceLength = raceLength - stepLength;

                if (raceLength <= 0) {
                    System.out.println(this.getName() + "到终点了");
                    if (raceCallback != null) {
                        raceCallback.win(this);
                    }
                }

                System.out.println(this.getName() + "跑了" + stepLength + "米，还剩" + raceLength + "米");

                if (raceLength % 2 == 0) {
                    Thread.sleep(1000);
                }
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        } finally {

        }
    }


}

乌龟

public class Tortoise extends Animal {

    public Tortoise() {
        setName("Tortoise");
    }

    @Override
    public void running() {
        try {
            while (!Thread.currentThread().isInterrupted()) {

冠军回调类

public class StopOtherCallback implements RaceCallback {

    List<Animal> animals;

    public StopOtherCallback() {
        animals = new ArrayList<Animal>();
    }

    public StopOtherCallback addAnimals(Animal animal) {
        animals.add(animal);
        return this;
    }

    public void win(Animal winer) throws InterruptedException {
        for (Animal animal : animals) {
            if (animal.equals(winer)) {// 请注意，这里需要把冠军放在最后结束，否则冠军自己结束线程之后无法通知其他赛跑人员
                continue;
            }
            //animal.interrupt();  //请注意，这里使用interrupt是无法停止线程的，因为线程可能处于sleep子线程中
            animal.stop();
        }
        //winer.interrupt();
        winer.stop();
    }
}

开始赛跑

public class Race {

    public static void main(String[] args) {
        Rabbit rabbit = new Rabbit();
        Tortoise tortoise = new Tortoise();

        StopOtherCallback stopOtherCallback = new StopOtherCallback();
        stopOtherCallback.addAnimals(rabbit).addAnimals(tortoise);

        rabbit.raceCallback = stopOtherCallback;
        tortoise.raceCallback = stopOtherCallback;

        rabbit.start();
        tortoise.start();
    }
}

5. 实例五、生产者–消费者模式

public class ProducerConsumer {
    /**
     * 管道保存一条消息，并且有标志确定管道是否有消息。
     * 管道有两种方法，分别是 take() and put().
     * put()在没有消息时放置消息, take()在有消息时接收消息.
     */
    public static class Pipe {
        private String message;
        private boolean empty = true;
        public synchronized void put(String message) {
            while (!empty) {
                try {
                    wait();
                } catch (InterruptedException e) {
                    System.out.println(e);
                }
            }
            empty = false;
            this.message = message;
            notifyAll();
        }
        public synchronized String take() {
            while (empty) {
                try {
                    wait();
                } catch (InterruptedException e) {
                    System.out.println(e);
                }
            }
            empty = true;
            notifyAll();
            return this.message;
        }
    }

    /**
     * 生产者向管道发送消息，并在消耗前一消息时放入下一个消息。
     */
    public static class Producer implements Runnable {
        private Pipe pipe;
        public Producer(Pipe pipe) {
            this.pipe = pipe;
        }
        public void run() {
            String importantInfo[] = {
                    "A baby girl was born.",
                    "The baby girl was named Mary.",
                    "Mary learned how to walk at age 1.",
                    "Mary learned how to speak at age 2.",
                    "Mary went to kindergarden at the age 3."
            };
            Random random = new Random();
            for (int i = 0; i < importantInfo.length; i++) {
                pipe.put(importantInfo[i]);
                try {
                    Thread.sleep(random.nextInt(5000));
                } catch (InterruptedException e) {
                    System.out.println(e);
                }
            }
            pipe.put("DONE");
        }
    }

    /**
     * 消费者在管道中存在消息时消费该消息。
     */
    public static class Consumer implements Runnable {
        private Pipe pipe;
        public Consumer(Pipe pipe) {
            this.pipe = pipe;
        }
        public void run() {
            Random random = new Random();
            for (String message = pipe.take(); !message.equals("DONE"); message = pipe.take()) {
                System.out.println("MESSAGE: " + message);
                try {
                    Thread.sleep(random.nextInt(5000));
                } catch (InterruptedException e) {
                    System.out.println(e);
                }
            }
        }
    }

    /**
     * 开始制作人和消费者演示
     * @param args
     */
    public static void main(String[] args) {
        Pipe drop = new Pipe();
        (new Thread(new Producer(drop))).start();
        (new Thread(new Consumer(drop))).start();
    }
}
