ArrayDeque双端队列

  • Java
0 views
2018年 07月30日

基本介绍

ArrayDeque是Resizable-array的实现。最多可以存储2^31 - 1 个元素。不支持多线程访问，不支持空元素操作。

ArrayDeque既可以实现堆栈的基本功能，也可以实现队列的基本功能。同时，当用作堆栈时，这个类的速度可能比堆栈快，当用作队列时，它比LinkedList更快。 大多数ArrayDeque操作都是在平摊的常数时间内运行的。

和大多数的集合迭代相似，迭代器也是快速失败的，快速迭代器会以最大努力抛出ConcurrentModificationException来保证程序正常运行。同样，使用ConcurrentModificationException来确定程序的正确性是错误的。

    /**

构造方法

    /**
     * 默认情况下，创建初始化容量为16的数组。
     */
    public ArrayDeque() {
        elements = new Object[16];
    }

    /**
     * 使用一个指定的容量来初始化数组。（由于数组容量是2的幂次数，所以如果这里输入9，最终初始化容量为16。）
     */
    public ArrayDeque(int numElements) {
        allocateElements(numElements);
    }

    /**
     * 使用一个集合来初始化数组。先创建输入，然后执行复制操作。
     */
    public ArrayDeque(Collection<? extends E> c) {
        allocateElements(c.size());
        addAll(c);//本方法来自AbstractCollection，具体实现是调用自身的add方法
    }

上面提到了allocateElements方法，看一下具体实现：

private void allocateElements(int numElements) {
        int initialCapacity = MIN_INITIAL_CAPACITY;
        // 如果初始化最小容量小于8，那么使用最小容量8来初始化数组
        if (numElements >= initialCapacity) {
            //下面几部操作实现获取 >= 当前值的最小的2的幂次数，具体操作就是把低位全都补全成1。
            initialCapacity = numElements;
            initialCapacity |= (initialCapacity >>>  1);
            initialCapacity |= (initialCapacity >>>  2);
            initialCapacity |= (initialCapacity >>>  4);
            initialCapacity |= (initialCapacity >>>  8);
            initialCapacity |= (initialCapacity >>> 16);
            initialCapacity++;

            if (initialCapacity < 0)   // 如果数值超过或者等于2^31，赋值为2^30（最大值）
                initialCapacity >>>= 1;
        }
        elements = new Object[initialCapacity];
    }

基本方法

基本的插入和获取方法是addFirst、addLast、pollFirst、poolLast，其他方法大都依赖与此。

添加

    /**
     * 在头部插入指定的元素（注意，一个空的、长度为16的数组，插入的第一个头部元素是在下标为 15的位置）
     */
    public void addFirst(E e) {
        //如果元素为空，抛出异常
        if (e == null)
            throw new NullPointerException();
        //插入一个元素在头部的前一个位置，请注意看最前面的图形解释，head增长的初始方向是下标减小的方向（这不是绝对的，因为head和tail会以一个圆环的形式轮转）
        elements[head = (head - 1) & (elements.length - 1)] = e;
        //如果head和tail相等了，表示元素已满，需要扩容
        if (head == tail)
            doubleCapacity();
    }

    /**
     * 插入指定的元素到尾部。这个方法和 add 方法基本一致。
     */
    public void addLast(E e) {
        //如果插入元素为空，抛出异常
        if (e == null)
            throw new NullPointerException();
        //tail位置存储目标元素，同时tail + 1，如果head和tail相等，表示元素已满，进行扩容操作
        elements[tail] = e;
        if ( (tail = (tail + 1) & (elements.length - 1)) == head)
            doubleCapacity();
    }

具体的扩容操作如下：

    //只有当head和tail因为某些操作导致两者相等的时候才会调用
    private void doubleCapacity() {
        //结合最上面的图片的情况一来分析，假设 tail == head == 10
        assert head == tail;
        int p = head;  // 10
        int n = elements.length; // 16
        int r = n - p; // 6 表示从head开始到结尾的元素个数
        int newCapacity = n << 1; // 容量 * 2
        if (newCapacity < 0) // 如果超过 2^30，也就是2^31为负值时，抛出异常
            throw new IllegalStateException("Sorry, deque too big");
        Object[] a = new Object[newCapacity]; //创建新数组
        System.arraycopy(elements, p, a, 0, r); // 把源数组从 p 开始的位置拷贝到目标数组的0位置
        System.arraycopy(elements, 0, a, r, p);// 把源数组从0开始的位置拷贝到目标数组的r位置
        elements = a;
        head = 0;//head设置为0，tail设置为n
        tail = n;
    }

获取&删除

    public E pollFirst() {
        int h = head;
        @SuppressWarnings("unchecked")
        E result = (E) elements[h]; //获取头元素
        // 如果队列为空，头元素为空
        if (result == null)
            return null;
        elements[h] = null;     // 清除头元素
        head = (h + 1) & (elements.length - 1); //头元素位置向着减小的方向移动一个
        return result;
    }

    public E pollLast() {
        int t = (tail - 1) & (elements.length - 1);
        @SuppressWarnings("unchecked")
        E result = (E) elements[t]; //获取尾元素
        if (result == null) //队列为空时，尾元素为空
            return null;
        elements[t] = null; //清除尾元素
        tail = t; //尾元素位置向着减小的方向移动一个
        return result;
    }

removeFirstOccurrence和removeLastOccurrence分别执行从头部和尾部查找第一次出现的元素并进行删除的操作：

    /**
     * 删除第一次出现的指定的元素。如果元素不存在，那么数组不会发生变化。
     */
    public boolean removeFirstOccurrence(Object o) {
        if (o == null)
            return false;
        int mask = elements.length - 1;
        int i = head; //获取第一个元素位置
        Object x;
        while ( (x = elements[i]) != null) {//依次比较元素是否相同
            if (o.equals(x)) {
                delete(i);
                return true;
            }
            i = (i + 1) & mask;//向着头元素删除的方向移动
        }
        return false;
    }

    /**
     * 删除第一次出现的指定元素，如果不存在，数组不变。
     */
    public boolean removeLastOccurrence(Object o) {
        if (o == null)
            return false;
        int mask = elements.length - 1;
        int i = (tail - 1) & mask;//获取最后一个元素
        Object x;
        while ( (x = elements[i]) != null) {//依次比较
            if (o.equals(x)) {
                delete(i);
                return true;
            }
            i = (i - 1) & mask;//向着尾元素删除的方向移动
        }
        return false;
    }

删除元素delete的操作稍微复杂一点儿：

private boolean delete(int i) {
        //校验不变的信息-->具体请看下面方法中的注释内容
        checkInvariants();
        //获取所有元素
        final Object[] elements = this.elements;
        final int mask = elements.length - 1;
        final int h = head;
        final int t = tail;
        //头元素前和尾元素后
        final int front = (i - h) & mask;
        final int back  = (t - i) & mask;

        // 如果head <= i < tail，i位置不会存在元素，抛出异常
        if (front >= ((t - h) & mask))
            throw new ConcurrentModificationException();

        // 下面的操作时为了保证移动最少的元素
        if (front < back) {
            //移动front的元素
            if (h <= i) {
                System.arraycopy(elements, h, elements, h + 1, front);
            } else { // Wrap around
                System.arraycopy(elements, 0, elements, 1, i);
                elements[0] = elements[mask];
                System.arraycopy(elements, h, elements, h + 1, mask - h);
            }
            elements[h] = null;
            head = (h + 1) & mask;
            return false;
        } else {
            //移动back的元素
            if (i < t) { // Copy the null tail as well
                System.arraycopy(elements, i + 1, elements, i, back);
                tail = t - 1;
            } else { // Wrap around
                System.arraycopy(elements, i + 1, elements, i, mask - i);
                elements[mask] = elements[0];
                System.arraycopy(elements, 1, elements, 0, t);
                tail = (t - 1) & mask;
            }
            return true;
        }
    }
    
    private void checkInvariants() {
        assert elements[tail] == null;//尾元素不能为空
        assert head == tail ? elements[head] == null : //head == tail 的情况下，头元素为空
            (elements[head] != null &&   //head != tail 的情况下，头元素不为空，尾元素不为空
             elements[(tail - 1) & (elements.length - 1)] != null);
        assert elements[(head - 1) & (elements.length - 1)] == null;//head前一个元素为空，表示队列不满
    }

新增

JDK1.8以后增加了Spliterator接口，ArrayDeque实现了这个并行迭代器：

    /**
     * 创建迭代器（late-binding & fail-fast）。
     */
    public Spliterator<E> spliterator() {
        return new DeqSpliterator<E>(this, -1, -1);
    }

    static final class DeqSpliterator<E> implements Spliterator<E> {
        private final ArrayDeque<E> deq;//数组对象
        private int index;  // 当前操作的序号值，初始为-1
        private int fence;  // 初始值-1

        /** 使用初始化的值来创建对象 */
        DeqSpliterator(ArrayDeque<E> deq, int origin, int fence) {
            this.deq = deq;
            this.index = origin;
            this.fence = fence;
        }

        // 强制初始化，如果fence为-1，设置fence为tail，index为head
        private int getFence() { 
            int t;
            if ((t = fence) < 0) {
                t = fence = deq.tail;
                index = deq.head;
            }
            return t;
        }

        public DeqSpliterator<E> trySplit() {
            int t = getFence(), h = index, n = deq.elements.length;
            if (h != t && ((h + 1) & (n - 1)) != t) {
                if (h > t)
                    t += n;
                int m = ((h + t) >>> 1) & (n - 1);
                return new DeqSpliterator<>(deq, h, index = m);
            }
            return null;
        }

        public void forEachRemaining(Consumer<? super E> consumer) {
            if (consumer == null)
                throw new NullPointerException();
            Object[] a = deq.elements;
            int m = a.length - 1, f = getFence(), i = index;
            index = f;
            while (i != f) {
                @SuppressWarnings("unchecked") E e = (E)a[i];
                i = (i + 1) & m;
                if (e == null)
                    throw new ConcurrentModificationException();
                consumer.accept(e);
            }
        }

        public boolean tryAdvance(Consumer<? super E> consumer) {
            if (consumer == null)
                throw new NullPointerException();
            Object[] a = deq.elements;
            int m = a.length - 1, f = getFence(), i = index;
            if (i != fence) {
                @SuppressWarnings("unchecked") E e = (E)a[i];
                index = (i + 1) & m;
                if (e == null)
                    throw new ConcurrentModificationException();
                consumer.accept(e);
                return true;
            }
            return false;
        }

        public long estimateSize() {
            int n = getFence() - index;
            if (n < 0)
                n += deq.elements.length;
            return (long) n;
        }

        @Override
        public int characteristics() {
            return Spliterator.ORDERED | Spliterator.SIZED |
                Spliterator.NONNULL | Spliterator.SUBSIZED;
        }
    }
